### Section 2 — PuzzleGenerator
Implement the constrained generation algorithm exactly as specified. All three guards (answer === 0 filter, subtraction b ≤ a constraint, dedup Set) MUST be present.

```javascript
// ── 2. PUZZLE GENERATOR ───────────────────────────────────────
// Pure function — no DOM access, no side effects. Testable in console.
const PuzzleGenerator = {
  generate(count = CONFIG.PUZZLE_COUNT) {
    const seen = new Set();
    const puzzles = [];

    while (puzzles.length < count) {
      const op = Math.random() < 0.5 ? '+' : '-';
      let a, b;

      if (op === '+') {
        // Constrain so a + b <= 19 (< MAX_RESULT=20)
        a = Math.floor(Math.random() * (CONFIG.MAX_RESULT - 1)); // 0–18
        b = Math.floor(Math.random() * (CONFIG.MAX_RESULT - a)); // 0–(19-a)
      } else {
        // Constrain so b <= a, guaranteeing a - b >= 0 (PUZZLE-03)
        a = Math.floor(Math.random() * CONFIG.MAX_RESULT);       // 0–19
        b = Math.floor(Math.random() * (a + 1));                 // 0–a
      }

      const answer = op === '+' ? a + b : a - b;

      // Skip zero-answer puzzles — young children may not recognize 0 as valid
      if (answer === 0) continue;

      // Dedup — reject if this exact expression already in the current set (PUZZLE-04)
      const key = `${a}${op}${b}`;
      if (seen.has(key)) continue;

      seen.add(key);
      puzzles.push({ operandA: a, operator: op, operandB: b, answer });
    }

    return puzzles;
  }
};
```

### Section 5 — UIRenderer
All four render methods plus updateTimer. Rules:
- All methods write to `this._app` (which is `document.getElementById('app')`) — never query the DOM elsewhere
- `renderQuestion` renders the full screen including a static numpad
- `updateTimer` targets ONLY `#timer-fill` — does NOT re-render the puzzle (prevents resetting typed answer)
- Event handlers in HTML attributes MUST call `GameController.*` methods only — no inline logic

```javascript
// ── 5. UI RENDERER ────────────────────────────────────────────
// All DOM writes. Receives data as arguments — never reads GameState directly.
const UIRenderer = {
  _app: document.getElementById('app'),

  renderStart() {
    this._app.innerHTML = `
      <div class="game-card screen-start">
        <div class="emoji-big">🔢</div>
        <h1>Math Adventure!</h1>
        <p>Can you solve <strong>10</strong> puzzles? 🌟</p>
        <button class="btn-primary" onclick="GameController.startGame()">▶ Play!</button>
      </div>`;
  },

  renderQuestion(puzzle, questionNumber, timeRemaining) {
    const pct = Math.round((timeRemaining / CONFIG.TIMER_SECONDS) * 100);
    this._app.innerHTML = `
      <div class="game-card screen-question">
        <div class="progress-row">
          <span>Question ${questionNumber} / ${CONFIG.PUZZLE_COUNT} ⭐</span>
          <span>Score: ${GameState.score}</span>
        </div>
        <div class="timer-bar-track">
          <div id="timer-fill" class="timer-bar-fill" style="width:${pct}%"></div>
        </div>
        <div class="puzzle-equation">${puzzle.operandA} ${puzzle.operator} ${puzzle.operandB} = ?</div>
        <div id="answer-display" class="answer-display">__</div>
        <div class="numpad">
          ${[1,2,3,4,5,6,7,8,9].map(n =>
            `<button class="numpad-btn" onclick="GameController.digitPress(${n})">${n}</button>`
          ).join('')}
          <button class="numpad-btn numpad-clear" onclick="GameController.backspace()">⌫</button>
          <button class="numpad-btn" onclick="GameController.digitPress(0)">0</button>
          <button class="numpad-btn numpad-submit" onclick="GameController.checkAnswer()">✅</button>
        </div>
      </div>`;
  },

  // Updates ONLY the timer bar — does NOT re-render the puzzle or numpad.
  // Called on every timer tick in Phase 3. Critical: never call renderQuestion() from a tick.
  updateTimer(timeRemaining) {
    const fill = document.getElementById('timer-fill');
    if (!fill) return;
    const pct = Math.max(0, Math.round((timeRemaining / CONFIG.TIMER_SECONDS) * 100));
    fill.style.width = pct + '%';
    // Color shift: green (>60%) → amber (30–60%) → red (<30%)
    if (pct > 60)      fill.style.background = 'var(--color-correct)';
    else if (pct > 30) fill.style.background = 'var(--color-timer-mid)';
    else               fill.style.background = 'var(--color-wrong)';
  },

  renderGameOver(score) {
    this._app.innerHTML = `
      <div class="game-card screen-gameover">
        <div class="emoji-big">😢</div>
        <h1>Game Over!</h1>
        <p>You solved <strong>${score}</strong> puzzle${score !== 1 ? 's' : ''}! 🌟</p>
        <button class="btn-primary" onclick="GameController.playAgain()">🔄 Try Again</button>
      </div>`;
  },

  renderResults(score) {
    const starCount = Math.max(1, Math.ceil(score / 2));   // 1–5 stars for scores 1–10
    const stars = '⭐'.repeat(starCount);
    this._app.innerHTML = `
      <div class="game-card screen-results">
        <div class="emoji-big">🎉</div>
        <h1>You Win!</h1>
        <p>All <strong>10</strong> puzzles solved! 🏆</p>
        <div class="stars">${stars}</div>
        <button class="btn-primary" onclick="GameController.playAgain()">🔄 Play Again</button>
      </div>`;
  }
};
```
